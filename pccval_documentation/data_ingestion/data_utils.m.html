import datetime
import glob
import importlib
import os
import shutil

import time
from typing import ClassVar
from typing import List

import pandas as pd

from data_ingestion import constants as c


def gen_folders(file_path: str) -> str:
    """
    Function to generate folders for the output of the validation tests.

    :return str outfolder_path: The path to the folder where the output of the
    validation tests is stored.
    """
    outfolder_path = os.path.join(file_path, 'outfolder')

    gen_option = ''
    if not os.path.exists(outfolder_path):
        os.mkdir(outfolder_path)
        print(('Folder {} created. Validation reports will be stored here.'
               ).format(outfolder_path))

    tmp_file_path = os.path.join(outfolder_path, 'tmp')
    if not os.path.exists(tmp_file_path):
        os.mkdir(tmp_file_path)

    else:
        os.chdir(os.path.join(outfolder_path, 'tmp'))
        csv_files = glob.glob('*.csv')
        if csv_files != []:
            csv_message = ('Folder {} already exists, containing the following csv files: {}'
                           ).format(outfolder_path, '\n* '.join(csv_files))
        else:
            csv_message = ('Folder {} already exists, but does not contain any files'
                           ).format(outfolder_path)

        gen_option = input(('{}\nDo you want to generate new csv files from PCC.xlsm? '
                            '(enter "Y" and hit enter to generate new files; to proceed '
                            'with the existing files simply hit enter.'
                            ).format(csv_message))
    if gen_option == 'Y':
        shutil.rmtree(outfolder_path)
        os.mkdir(outfolder_path)
    else:
        pass

    return gen_option, outfolder_path

def read_in_df(tab_name: str, path: str) -> pd.DataFrame:
    """
    Read in dataframe (df)

    Function to read in a tab from the PCC xls file for further analysis.

    Example use: read_in_df('../PCC.xlsm', 'ELEN')

    :param str path: Path to the csv file
    :param str tab_name: Name of the tab to be read in
    :return: pd.DataFrame tab_xls: The tab with relevant tmp
    """
    file_xls = pd.ExcelFile(path)

    tab_xls = pd.read_excel(file_xls, tab_name)

    return tab_xls

def gen_csv_files(path: str, outfolder: str) -> str:
    """
    Read in dataframe (df)

    Function to read in a tab from the PCC xls file for further analysis.

    Example use: read_in_df('../PCC.xlsm', 'ELEN')

    :param str path: Path to the csv file
    :return: pd.DataFrame tab_xls: The tab with relevant tmp
    :return: str folder_path: The path to the folder where the csv files
    for the PCC tmp frames are stored
    """
    # TODO change call on 'PCC' such that it can be used with any file
    folder_path = os.path.join(outfolder, 'tmp')
    print('Generating separate csv files for all PCC tmp frames '
          'at {}'.format(folder_path))

    if not os.path.exists(folder_path):
        os.makedirs(folder_path)

    file_xls = pd.ExcelFile(path)
    sheet_names = file_xls.sheet_names

    for sheet in sheet_names:
        if sheet in get_class_attributes(c.DFN):
            tab_xls = pd.read_excel(file_xls, sheet, header=0)
            tab_xls.to_csv(
                            os.path.join(folder_path, (sheet + '.csv')),
                            header=True,
                            sep=','
                            )
            print('CSV file for {} tmp frame generated'.format(sheet))
    return folder_path

def generate_output_folders():
    """
    Generate output folders.

    Function that generates folders (infolder and outputfolder).
    """
    valid_file = False
    while valid_file == False:
        file_path = input(
            'Please enter the full path to your the folder where you have '
            'stored your PCC.xlsm file: ')
        if os.path.exists(file_path):
            valid_file = True
        else:
            print('The path "{}" is not a valid path. Please '
                  'enter a valid, full path to the folder where you have '
                  'stored the PCC .xlsm file.'.format(file_path))

    os.chdir(file_path)
    raw_csv_file = glob.glob('*.xlsm')
    assert len(raw_csv_file) is not 0, ('Critical error: No PCC.xlsm file '
                                        'present in folder {}.'
                                        ).format(file_path)
    raw_csv_file = raw_csv_file[0]

    output_file_constant = raw_csv_file.split('.xlsm')[0] + '_analysis'

    gen_option, outfolder = gen_folders(file_path.replace(raw_csv_file, ''))

    return gen_option, file_path, output_file_constant, outfolder, raw_csv_file

def parse_date(text: str):
    """
    Parse date

    Function to convert a str date following the format '%Y' or '%d%b%Y' to
    datetime.date() format.

    :param str text: The text to be parsed into a date.
    :return datetime.date() date: Date object.
    """
    for fmt in ('%Y', '%d%b%Y'):
        try:
            date = datetime.datetime.strptime(text, fmt).date()
            return date
        except ValueError:
            pass
    raise ValueError('Date format not valid.')

def get_class_attributes(obj) -> List[str]:
    """
    Function to obtain the values of all attributes of a class.

    :param ClassVar obj: A tmp ingestion constants class
    :return: List[str] class_vals: A list containing the values of all
    attributes associated with the class.
    """
    class_objects = [attr for attr in dir(obj) if not
    callable(getattr(obj, attr)) and not attr.startswith('__')]

    class_vals = [getattr(obj, i) for i in class_objects]

    return class_vals

def gen_time() -> str:
    """
    Function to generate timestamp in string format (used for logging
    purposes)
    :return str current_time: Current time (format: '%Y-%m-%d %H:%M:%S')
    """
    current_time = datetime.datetime.fromtimestamp(time.time()
                                                   ).strftime('%Y-%m-%d %H:%M:%S')
    return current_time

def import_class(class_name: str,
                 data_path: str,
                 reporting_instance,
                 module_name: str='pcc_validation'):
    """
    Function to import a class by class name and module name

    :param str class_name: The name of the class.
    :param str module_name: The name of the module.
    :return Class instance: The class instance associated with the class and
    module name.
    """
    module = importlib.import_module(module_name +
                                     '.{}_validate'.format(
                                                            class_name.lower()
                                                        )
                                     )
    class_ = getattr(module, (class_name + 'Validate'))
    instance = class_(data_path, reporting_instance)
    return instance
