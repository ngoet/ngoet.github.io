# -*- coding: utf-8 -*-
"""
    tests.unit_test_utils
    ~~~~~~~~~~~~~
    Class of functions for unit tests

    :copyright: Â© 2019 Niels Goet @ PCC Project
"""
import logging
import os
from typing import List
import pandas as pd

from data_ingestion import constants as c
from data_ingestion import data_utils
from reporting import visualization_utils
from reporting.reporting_utils import Reporting, ReportingItem


class Validate:
    """
    Validate. Class of functions to run the validation tests for the PCC tmp.
    """

    def __init__(self, flag: List[str] = ['all'], gen_pdf: bool=False):
        # Obtain and check flags
        # self.flag = sys.argv[0]
        self.gen_pdf = gen_pdf
        self.flag = flag
        self.supported_flags = data_utils.get_class_attributes(c.DFN)

        # Get timestamp
        self.timestamp = data_utils.gen_time()

        assert any(i in ['all'] + self.supported_flags for i in self.flag), (
            c.RUNM.FLAG.format(['--all'].append(['--' + i for i in
                                                 self.supported_flags])))

        # Obtain tmp path and generate folders
        self.gen_option, self.file_path, _, self.outfolder, \
        self.raw_csv_file = data_utils.generate_output_folders()
        print('Results of the analysis will be stored in the '
              'following folder: {}'.format(self.outfolder))

        # Generate temp folder
        tmp_folder = os.path.join(self.outfolder, 'tmp')
        if not os.path.exists(tmp_folder):
            os.makedirs(tmp_folder)

        # Generate figures folder
        fig_folder = os.path.join(tmp_folder, 'figures')
        if not os.path.exists(fig_folder):
            os.makedirs(fig_folder)

        # Generate tables folder
        tab_folder = os.path.join(tmp_folder, 'tables')
        if not os.path.exists(tab_folder):
            os.makedirs(tab_folder)

        # Generate separate tmp frames
        if self.gen_option == 'Y':
            self.folder_path = data_utils.gen_csv_files(os.path.join(self.file_path,
                                                                     self.raw_csv_file),
                                                        self.outfolder)

        # Instantiate the logging instance
        log_path = os.path.join(self.outfolder,
                                'log[{}].txt'.format(self.timestamp))
        self.log = logging.basicConfig(filename=log_path,
                                       level=logging.DEBUG)

        # Instantiate the reporting instance
        self.reporting_instance = Reporting(
            validation_report_file_path=self.file_path,
            log_file=self.log
        )


    @staticmethod
    def _run_class_functions(x, reporting_item):
        """
        Function that dynamically calls in functions associated with a class,
        and executes them.

        :param Class x: Class instance from which to call functions
        """
        public_method_names = [method for method in dir(x) if
                               callable(getattr(x, method)) if
                               not method.startswith(
                                   '_')]

        # Add information on available validation tests
        reporting_item.performed_validation_tests = public_method_names

        # Execute all public methods in the class
        for method in public_method_names:
            getattr(x, method)(reporting_item)  # call

    def _run_class_validation(self, df_id: str):
        """
        Function to call in a class instance based on its name and execute
        by using :func Validate.__run_class_functions:

        :param str df_id: The name associated with the class
        :return: ReportingItem reporting_item: An md report based on the
        validation tests associated with the relevant class
        """
        print('{}: Starting validation tests for {}'.format(
            data_utils.gen_time(), df_id))

        # Instantiate reporting item and add df level
        reporting_item = ReportingItem()
        reporting_item.df_level = df_id

        self.data_path = os.path.join(self.outfolder, 'tmp')

        CLASS_instance = data_utils.import_class(class_name=df_id,
                                                 data_path=self.data_path,
                                                 reporting_instance=self.reporting_instance)
        self._run_class_functions(CLASS_instance, reporting_item)

        return reporting_item

    def data_checker(self, df_name: str):
        """
        Checks if there is any date in the data frame. Use in
        :func run_validation: to determine if functions should be executed
        for the class instantiation.

        :param str df_name: The name of the data frame for which the
        data check needs to be performed
        """
        # Read in the file
        assert df_name in data_utils.get_class_attributes(c.DFN), \
            'Data frame with name {} is not supported.'.format(df_name)

        df = pd.read_csv(os.path.join(os.path.join(self.outfolder, 'tmp'),
                                      df_name + '.csv'
                                      ), header=0, nrows=1, index_col=0)

        if len(df.index) > 0:
            return True
        else:
            return False

    def run_validation(self):
        """
        Function to run validation tests based on flag options. Depends on
        :func Validate.__run_class_validation:
        """
        print('{}: Starting validation tests'.format(data_utils.gen_time()))

        reporting_objects = []
        if self.flag == ['all']:
            self.flag = data_utils.get_class_attributes(c.DFN)

            # Run class functions for all classes
        for df_id in self.flag:
            if self.data_checker(df_id):
                report = self._run_class_validation(df_id)
                reporting_objects.append(report)
            else:
                print('Test will not be performed for {}, since it is empty.'.format(df_id))


        # Write out results
        # Add summary stats across all reported modules
        summary_stats = pd.DataFrame(
            {'Data frame': [i.df_level for i in reporting_objects],
             'Total potential errors': [i.n_errors for i in reporting_objects]}
        )
        summary_stats.loc[:, 'Proportion of total errors'] = summary_stats['Total potential errors'] / summary_stats['Total potential errors'].sum()

        self.reporting_instance.add_header(level=3, header_text='Summary statistics', write_out= True)
        self.reporting_instance.txt_to_md(self.reporting_instance.add_table(summary_stats))

        # Generate figure
        if len(reporting_objects) > 1:
            fig_path = visualization_utils.gen_bar_chart(path=self.data_path, vars=['Data frame', 'Proportion of total errors'], fig_name='summary_bar_chart.png',
                                          df=summary_stats)
                                          # log=self.log)

            self.reporting_instance.txt_to_md('\n![]({})\n'.format(fig_path))

        # Write out report objects
        for i in reporting_objects:
            # Add class header
            self.reporting_instance.txt_to_md(self.reporting_instance.add_header(header_text='{} - Validation Tests Report'.format(i.df_level), level=2))

            error_report = pd.DataFrame(
                {'test': [x for x in i.error_dict.keys()],
                 'errors': [x for x in i.error_dict.values()]})

            # Add number of errors
            self.reporting_instance.txt_to_md(
                '\nTotal number of potential errors identified: {}\n'.format(
                    error_report['errors'].sum()))

            # Add supported validation tests
            self.reporting_instance.txt_to_md('\n The following validation tests were performed for {}:\n'.format(i.df_level))

            self.reporting_instance.txt_to_md(self.reporting_instance.add_table(error_report))

            # Add report
            self.reporting_instance.df_results_to_md(i)

        # Convert to pdf
        # if self.gen_pdf:
