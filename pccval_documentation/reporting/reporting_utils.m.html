#!/usr/bin/python
# -*- coding: utf-8 -*-
"""
    reporting.reporting
    ~~~~~~~~~~~~~
    Functions for logging the results of the PCC validation tests.

    :copyright: © 2019 Niels Goet @ PCC Project
"""
from os import path, mkdir
from data_ingestion import data_utils
from tabulate import tabulate
import pandas as pd
import os
from typing import List
import logging
from reporting import visualization_utils
from data_ingestion import constants as c


def italicize(txt: str, whiteline: str = '\n'):
    txt = ('{0}*' + ' '.join(txt.replace('\n', '').split()) + '*{0}').format(whiteline)

    return txt


class ReportingItem:
    """
    Class for the ReportingItem object. One such object is instantiated for
    each class of validation tests (i.e. one per PCC data frame), and
    populated with validation test results.
    """
    def __init__(self,
                 df_level=None,
                 entry_md_text='',
                 error_dict={},
                 performed_validation_tests=None,
                 n_errors=0):

        self.df_level = df_level
        self.entry_md_text = entry_md_text
        self.error_dict = error_dict
        self.performed_validation_tests = performed_validation_tests
        self.n_errors = n_errors


class Reporting:
    """
    Class of functions to generate a validation report for all tests in the
    pcc_val package

    Class instantiation creates a log file that includes the validation
    report in markdown format, and a folder that contains figures. The
    filename convention for this file is: ppc_validation_report[timestamp].md
    """
    def __init__(self,
                 log_file,
                 validation_report_file_path: str = path.join(os.getcwd(),
                                                              'tests')
                 ):

        self.validation_report_file_path = validation_report_file_path
        self.write_mode = 'ab'
        self.log = log_file

        assert path.exists(self.validation_report_file_path), (
            'The path {} does not exist. '
            'Cannot generate validation '
            'report.'
            ).format(self.validation_report_file_path)

        # Generate a folder to store figures from the validation exercise
        fig_path = path.join(self.validation_report_file_path, 'figures')
        if not path.exists(fig_path):
            mkdir(fig_path)

        # Create log file
        timestamp = data_utils.gen_time()
        file_name = 'pcc_validation_report{}.md'.format(timestamp.replace('-', '_'
                                                                ).replace(':', '_'
                                                                ).replace(' ','_'))

        self.validation_report_file_path = path.join(self.validation_report_file_path, file_name)

        with open(self.validation_report_file_path, self.write_mode) as log_file:
            log_file.write(c.REP.INTRO_TEXT.format(timestamp).encode('utf-8'))

    def txt_to_md(self, txt: str):
        """
        Adds a text entry to the log file.

        :param str txt: Any text or md-formatted entry.
        """
        with open(self.validation_report_file_path, self.write_mode) as log_file:
            log_file.write(txt.encode('utf-8'))

    def add_entry(self, txt: str, reporting_item: ReportingItem):
        """
        Adds a text entry to the log file.

        :param str txt: Any text or md-formatted entry.
        """
        reporting_item.entry_md_text.append(txt)

    def df_results_to_md(self, reporting_item: ReportingItem):
        with open(self.validation_report_file_path,
                  self.write_mode) as log_file:
            log_file.write(reporting_item.entry_md_text.encode('utf-8'))
        log_file.close()

    def list_to_md(self,  items: List, introductory_text: str = None,
                   list_st: str = '* {}\n'):
        with open(self.validation_report_file_path,
                  self.write_mode) as log_file:
            if introductory_text is not None:
                log_file.write(introductory_text.encode('utf-8'))

            for x in items:
                log_file.write(list_st.format(x).encode('utf-8'))

        log_file.close()

    def add_header(self, level: int, header_text: str,
                   write_out: bool = False):

        if not isinstance(level, int):
            raise ValueError('Level should be an integer value.')

        if not isinstance(header_text, str):
            raise ValueError('The header text should be a string.')

        assert level in [2, 3, 4], ('Header level cannot be smaller '
                                    'than 2 or bigger than 4')

        entry = ('\n {} {}'.format('#' * level, header_text))
        if write_out:
            self.txt_to_md(entry)

        return entry

    @staticmethod
    def add_table(df: pd.DataFrame, collapsible: bool=False):
        """
        Adds a table in markdown format to the log file.

        :param pd.DataFrame df: A pandas df that contains the tmp that
        needs to be included in the md.
        """
        data_path = os.path.join(os.getcwd(), 'tables')
        if not os.path.exists(data_path):
            os.mkdir(data_path)

        files = [os.path.join(data_path, f) for f in os.listdir(data_path)
                if os.path.isfile(os.path.join(data_path, f))]

        n_files = len([i for i in enumerate(files)])
        table_name = os.path.join(data_path, 'table_{}'.format(n_files + 1))

        md_table = tabulate(df, tablefmt='pipe', headers='keys',
                            showindex=False)

        df.to_csv(table_name)
            # # Add collapsible format
            # if collapsible:
            #     md_table = '
Click here to view the results table for this validation test\n{}\n
'.format(md_table)

        if len(df.index) <= 50:
            return '\n {} \n\n![Download table as .csv]({})\n'.format(md_table, table_name)

        else:
            return '\n![Too many observations - Download table as .csv]({})\n'.format(table_name)
        # self.add_entry('\n {} \n'.format(md_table))

    def add_figure(self, fig_path: str,
                        y_vals: List[int],
                        x_vals: List[int],
                        y_lab: str,
                        x_lab: str,
                        plt_title: str,
                        fig_type = str):
        """
        Adds a figure reference to the md file (![]{file_path}).
        :param str fig_path: The path to the image.
        """
        assert fig_type in c.FIGS.SUPPORTED, ('Figure type {} not supported.'
                                              ).format(fig_type)

        public_method_names = [method for method in
                               dir('reporting.visualization_utils') if
                               callable(getattr(fig_type, method)) if
                               not method.startswith(
                                   '_')]
        for method in public_method_names:
            if fig_type in method:
                getattr('reporting.visualization_utils', method)(fig_path,
                                                                   y_vals,
                                                                   x_vals,
                                                                   y_lab,
                                                                   x_lab,
                                                                   plt_title,
                                                                   self.log)

        self.add_entry('![]({})'.format(fig_path))
