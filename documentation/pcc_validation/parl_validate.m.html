#!/usr/bin/python
# -*- coding: utf-8 -*-
"""
    pcc_validation.parl_validate
    ~~~~~~~~~~~~~
    Class of functions to test the PARL df.

    :copyright: Â© 2019 Niels Goet @ PCC Project
"""
import datetime
import sys
import os
import pandas as pd

from data_ingestion import data_utils
from data_ingestion import constants as c
from pcc_validation import validation_utils
from reporting import reporting_utils
from reporting.reporting_utils import Reporting, ReportingItem

sys.path.append(os.path.dirname(os.path.realpath(__file__)) + '/./')
sys.path.append(os.path.dirname(os.path.realpath(__file__)) + '/../')

class PARLValidate:
    """
    PARL Validate

    Class that contains validation tests for the Parliament Data Frame (PARL).
    """

    def __init__(self, data_path: str, reporting_instance: Reporting):
        self.reporting_instance = reporting_instance
        self.data_path = data_path
        self.validation_level = c.DFN.PARL
        self.test_file = pd.read_csv(os.path.join(self.data_path,
                                                  self.validation_level +
                                                  '.csv'), header=0)

        self.primary_key = list(self.test_file.columns)[1]

        self.poli_df = pd.read_csv(os.path.join(self.data_path,
                                                c.DFN.POLI +
                                                '.csv'), header=0)

        self.parl_df = pd.read_csv(os.path.join(self.data_path,
                                                c.DFN.PARL +
                                                '.csv'), header=0)

        print(c.LOGM.DF_RUN_MESSAGE.format(self.validation_level))

    def validate_primary_keys(self, reporting_item: ReportingItem):
        """
        Tests whether the primary ids are unique, i.e. non-repeating.
        """
        validation_utils.check_primary_keys(df=self.test_file,
                                            reporting_instance=self.reporting_instance,
                                            extra_text=(
                                                reporting_utils.italicize(str(
                                                    self.validate_primary_keys.__doc__))),
                                            reporting_item=reporting_item
                                            )

    def test_leg_period_ids(self, reporting_item: ReportingItem):
        """
        Function to check if the legislative period identifiers
        are consecutive and increasing for each parliament
        """
        problem_dfs = []
        for parl in self.test_file[c.PARL.ASSEMBLY_ABB].unique():
            df_parl = self.test_file[self.test_file[c.PARL.ASSEMBLY_ABB] == parl]

            try:
                # Check if numbers for leg_period are consecutive
                condition = validation_utils.check_consecutive(df_parl[c.PARL.LEG_PERIOD].values)
                problem_dfs.append(df_parl[condition])
            except:
                pass

        reporting_item.entry_md_text += self.reporting_instance.add_header(
            level=3,
            header_text=(
                self.test_leg_period_ids.__name__)
        )

        # Add docstrings to the .md report
        reporting_item.entry_md_text += (
            reporting_utils.italicize(
                str(
                    self.test_leg_period_ids.__doc__.split(':param')[0]
                )
            )
        )

        n_problems = 0
        if len(problem_dfs) > 0:
            concat_df = pd.concat(problem_dfs)
            n_problems = len(concat_df.index)
            reporting_item.entry_md_text += c.REP.LEGISLATIVE_PERIOD_IDS.format(
                len(concat_df.index),
                str(concat_df[c.PARL.PARLIAMENT_ID].unique())
            )
            
            reporting_item.entry_md_text += self.reporting_instance.add_table(concat_df)

        else:
            reporting_item.entry_md_text += c.REP.PASS

        reporting_item.error_dict.update({self.test_leg_period_ids.__name__: n_problems})
        reporting_item.n_errors += n_problems

    def test_start_end_date(self, reporting_item):
        """
        Check start and end date

        Checks if the start date of the legislative period lies before the
        end date of the legislative period.
        """
        def catch(x):
            try:
                val = datetime.datetime.strptime(x, '%d%b%Y')

            except:
                val = datetime.datetime.strptime(x, '%Y')

            return val

        start_dates = [catch(i.split('[[')[0]) for i in self.test_file[c.PARL.LEG_PERIOD_START]]
        end_dates = [catch(i.split('[[')[0]) for i in self.test_file[c.PARL.LEG_PERIOD_END]]
        condition = [x > y for x,y in zip(start_dates, end_dates)]
        problem_cases = self.test_file[condition]

        reporting_item.entry_md_text += self.reporting_instance.add_header(
            level=3,
            header_text=(
                self.test_start_end_date.__name__)
        )

        # Add docstrings to the .md report
        reporting_item.entry_md_text += (
            reporting_utils.italicize(
                str(
                    self.test_start_end_date.__doc__.split(':param')[0]
                )
            )
        )

        if len(problem_cases.index) > 0:
            reporting_item.entry_md_text += c.REP.GENERIC.format(len(problem_cases.index))
            reporting_item.entry_md_text += self.reporting_instance.add_table(problem_cases)

        else:
            reporting_item.entry_md_text += c.REP.PASS

        reporting_item.error_dict.update({self.test_start_end_date.__name__: len(problem_cases.index)})
        reporting_item.n_errors += len(problem_cases.index)

    def test_date_and_leg_ids(self, reporting_item):
        """
        Checks whether the dates and legislative period identifiers correspond,
        i.e. if leg_period at t-1 vs t, then the leg_period_start at t-1 should
        also be smaller. Applies to both leg_period_start and leg_period_end.
        """
        df = self.test_file[[c.PARL.PARLIAMENT_ID,
                             c.PARL.LEG_PERIOD,
                             c.PARL.LEG_PERIOD_START,
                             c.PARL.LEG_PERIOD_END]
                            ].sort_values([c.PARL.PARLIAMENT_ID,
                                          c.PARL.LEG_PERIOD])
        # Split by identifier
        ids = [i.split('_')[0]+ '-' + i.split('_')[1] for i in
               df[c.PARL.PARLIAMENT_ID].unique()]
        df['ids'] = ids
        dfs = [i[1].reset_index() for i in df.groupby('ids')]

        incorrect_start_dates = {}
        incorrect_end_dates = {}
        for df_i in dfs:
            for i in range(1, len(df_i.index)):
                date0_start = data_utils.parse_date(df_i.loc[i - 1, :][c.PARL.LEG_PERIOD_START].split('\[')[0])
                date1_start = data_utils.parse_date(df_i.loc[i, :][c.PARL.LEG_PERIOD_START].split('\[')[0])
                date0_end = data_utils.parse_date(df_i.loc[i - 1, :][c.PARL.LEG_PERIOD_END].split('\[')[0])
                date1_end = data_utils.parse_date(df_i.loc[i, :][c.PARL.LEG_PERIOD_START].split('\[')[0])
                # Only check if the date is an integer
                if df_i.loc[i, :][c.PARL.LEG_PERIOD] in [str(i) for i in range(200)]:
                    leg_period0 = int(df_i.loc[i-1, :][c.PARL.LEG_PERIOD])
                    leg_period1 = int(df_i.loc[i-0, :][c.PARL.LEG_PERIOD])
                    if not ((leg_period0 < leg_period1) &
                            (date0_start < date1_start)):
                        incorrect_start_dates.update(
                                {df_i.loc[i, :][c.PARL.PARLIAMENT_ID]:
                                     ('{0} t = {2}: ; {0} = t-1: {3}; '
                                      '{1} t = {4}: ; {1} = t-1: {5}'
                                      ).format(
                                                c.PARL.LEG_PERIOD_START,
                                                c.PARL.LEG_PERIOD,
                                                date1_start,
                                                date0_start,
                                                leg_period1,
                                                leg_period0
                                            )
                                }
                            )

                    if not ((leg_period0 < leg_period1) &
                            (date0_end <= date1_end)):
                        incorrect_end_dates.update(
                                {df_i.loc[i, :][c.PARL.PARLIAMENT_ID]:
                                     ('{0} t = {2}: ; {0} = t-1: {3}; '
                                      '{1} t = {4}: ; {1} = t-1: {5}'
                                      ).format(
                                                c.PARL.LEG_PERIOD_END,
                                                c.PARL.LEG_PERIOD,
                                                date1_end,
                                                date0_end,
                                                leg_period1,
                                                leg_period0
                                            )
                                }
                            )
        reporting_item.entry_md_text += self.reporting_instance.add_header(
            level=3,
            header_text=(
                self.test_date_and_leg_ids.__name__)
        )

        # Add docstrings to the .md report
        reporting_item.entry_md_text += (
            reporting_utils.italicize(
                str(
                    self.test_date_and_leg_ids.__doc__.split(':param')[0]
                )
            )
        )

        incorrect_dates_df = pd.concat(
            [
                pd.DataFrame.from_records(incorrect_start_dates, index=range(len(incorrect_start_dates))).transpose(),
                pd.DataFrame.from_records(incorrect_end_dates, index=range(len(incorrect_end_dates))).transpose()]
            ).reset_index().rename(columns={'index': c.PARL.PARLIAMENT_ID, 0: 'Error message'})

        if len(incorrect_dates_df.index) > 0:
            reporting_item.entry_md_text += c.REP.GENERIC.format(len(incorrect_dates_df.index))
            reporting_item.entry_md_text += self.reporting_instance.add_table(incorrect_dates_df)

        else:
            reporting_item.entry_md_text += c.REP.PASS

        reporting_item.error_dict.update({self.test_date_and_leg_ids.__name__: len(incorrect_dates_df.index)})
        reporting_item.n_errors += len(incorrect_dates_df.index)

    def validate_date_formats(self, reporting_item):
        """
        Tests whether the date variables included in the tmp frame follow
        the specified format (%d%b%Y).
        """
        validation_utils.check_dates(
            df=self.test_file,
            primary_id_col=self.primary_key,
            date_columns=[c.PARL.LEG_PERIOD_START, c.PARL.LEG_PERIOD_END],
            reporting_instance=self.reporting_instance,
            extra_text=(reporting_utils.italicize(
                str(self.validate_date_formats.__doc__))),
            reporting_item=reporting_item
        )

def test_start_end_date(self, reporting_item):
    """
    Check start and end date
    Checks if the start date of the legislative period lies before the
    end date of the legislative period.
    """
    def catch(x):
        try:
            val = datetime.datetime.strptime(x, '%d%b%Y')
        except:
            val = datetime.datetime.strptime(x, '%Y')
        return val
    start_dates = [catch(i.split('[[')[0]) for i in self.test_file[c.PARL.LEG_PERIOD_START]]
    end_dates = [catch(i.split('[[')[0]) for i in self.test_file[c.PARL.LEG_PERIOD_END]]
